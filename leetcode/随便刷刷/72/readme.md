            - 状态转移方程
                dp[i][j] 跟前面状态的关系 最优的 
                    dp[i-1][j-1]
                    dp[i-1][j]
                    dp[i][j-1]

                    for((i)) {
                        for((j)) {
                            if(w1[i] == w1[j]) {
                                dp[i][j] = dp[i-1][j-1]
                            }else if(w1[i] != w1[j]) {
                                // 加 删除的反状态
                                // 删除 dp[i-1][j] + 1  dp[i][j-1] + 1
                                // 换 dp[i-1][j-1] + 1
                                dp[i][j] = Math.min(dp[i - 1][j - 1] + 1, dp[i - 1][j] + 1, dp[i][j - 1] + 1) 
                            }
                        }
                    }
